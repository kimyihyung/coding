1부터 100까지 출력하기

1.1 while문을 이용해서 1부터 100까지 출력하시오

for( let i=1; i<=100; i++){
    if ( i % 5 == 0){
        document.write("<span style='color: blue'>"+i+"</span>");  //5의 배수 출력
    }
    if ( i % 7 == 0 ){
    document.write("<span style='color: red'>"+i+"</span>");  //7의 배수 출력
    }
}
